import { Component, OnInit } from '@angular/core';
import { environment } from 'src/environments/environment';
import { FormGroup, FormBuilder, Validators } from '@angular/forms';
import { HttpClient, HttpHeaders, HttpBackend } from '@angular/common/http';
import { Router } from '@angular/router';
import { SnackBarService } from '../../services/snack-bar.service';
@Component({
  selector: 'app-token-page',
  templateUrl: './token-app.component.html',
  styleUrls: ['./token-app.component.scss']
})
export class TokenAppComponent implements OnInit {
  tokenFormGroup: FormGroup;
  private httpClient: HttpClient;
  enableCaptcha = false;
  constructor(private snackBarService: SnackBarService, private formBuilder: FormBuilder, handler: HttpBackend) {
    this.httpClient = new HttpClient(handler);
  }
 
  ngOnInit() {
    this.tokenFormGroup = this.formBuilder.group({
      rim: ['1443678', [Validators.required, Validators.minLength(7)]],
      pin: ['2233', [Validators.required, Validators.minLength(4)]]
    });
  }
  genJWT() {
    const data = 'grant_type=password&client_id=AuthenticatorClient&username='
    + this.tokenFormGroup.get('rim').value + '&password=' +
    + this.tokenFormGroup.get('pin').value + '&scope=read&mode=rimepin';
    if (this.tokenFormGroup.get('rim').value && this.tokenFormGroup.get('pin').value) {
      this.callCIAM(data).subscribe(
        res => {
          if (res && res.status === 200) {
          localStorage.setItem('access_token', res.body['access_token']);
          window.location.href = '';
          }
        },
        err => {
          this.snackBarService.showSnackBar({
                showSnackBar: true, message: {
                    msgType: 'error', msgText: err['message']
                }
            });
        },
      );
    }
  }
 
  callCIAM(data: any) {
    const header = new HttpHeaders().set('Content-Type', 'application/x-www-form-urlencoded')
    .set('X-IBM-Client-Id', environment.IBM_CLIENT_ID);
    return this.httpClient.post(encodeURI(environment.CIAM_API), data, {
      headers: header, observe: 'response', responseType: 'json' as any });
  }
 
  handleRecaptcha() {
    this.enableCaptcha = !this.enableCaptcha;
  }
 
  resolved(captchaResponse: string) {
    console.log(`${captchaResponse}`);
  }
}
