    }
  }

        || Number(amountValue < INSIDE_UAE_TRANSFER_TEXT.minimumTransferAmountCrossCurrency))) {

      FORM_CONTROL_AMOUNT.setErrors(
        { notValidAmount: { message: INSIDE_UAE_TRANSFER_TEXT.minimumAmountError } });
    }
  }
